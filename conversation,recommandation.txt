def discretize_for_bn(df, n_bins=5):
    out = df.copy()
    numeric_maps = {}

    cat_cols = [c for c in out.columns if c not in BN_NUMERIC_COLS]

    for c in cat_cols:
        out[c] = out[c].astype(str).fillna("MISSING")

    for c in BN_NUMERIC_COLS:
        s = pd.to_numeric(out[c], errors="coerce")
        s = s.fillna(s.median())

        try:
            cats = pd.qcut(s, q=n_bins, duplicates="drop")
        except Exception:
            nbins = max(2, min(n_bins, int(s.nunique())))
            cats = pd.cut(s, bins=nbins, duplicates="drop")

        out[c] = cats.astype(str).fillna("MISSING")

        # Mapping intervalle -> milieu pour revenir à du numérique après sampling
        numeric_maps[c] = {
            str(interval): float((interval.left + interval.right) / 2.0)
            for interval in cats.cat.categories
        }

    return out.astype(str), numeric_maps

def apply_notebook_style_preprocessing(df):
    df = df.copy()

    # TARGET
    if TARGET_COL in df.columns:
        df[TARGET_COL] = pd.to_numeric(df[TARGET_COL], errors="coerce").fillna(0).round().astype(int)

    # Label encoding binaire comme dans ton notebook
    le = LabelEncoder()
    for col in df.columns:
        if col == TARGET_COL:
            continue

        dtype_str = str(df[col].dtype)
        is_text_like = (
            pd.api.types.is_object_dtype(df[col])
            or pd.api.types.is_string_dtype(df[col])
            or dtype_str.startswith("category")
        )

        if is_text_like:
            vals = df[col].fillna("MISSING").astype(str)
            if vals.nunique(dropna=False) <= 2:
                le.fit(vals)
                df[col] = le.transform(vals)
            else:
                df[col] = vals

    # One-hot
    df = pd.get_dummies(df, dummy_na=False)

    # Reprise de la logique de ton notebook
    if "DAYS_EMPLOYED" in df.columns:
        df["DAYS_EMPLOYED_ANOM"] = df["DAYS_EMPLOYED"] == 365243
        df["DAYS_EMPLOYED"] = df["DAYS_EMPLOYED"].replace(365243, np.nan)

    if "DAYS_BIRTH" in df.columns:
        df["DAYS_BIRTH"] = df["DAYS_BIRTH"].abs()

    if {"AMT_CREDIT", "AMT_INCOME_TOTAL"}.issubset(df.columns):
        df["CREDIT_INCOME_PERCENT"] = df["AMT_CREDIT"] / df["AMT_INCOME_TOTAL"].replace(0, np.nan)

    if {"AMT_ANNUITY", "AMT_INCOME_TOTAL"}.issubset(df.columns):
        df["ANNUITY_INCOME_PERCENT"] = df["AMT_ANNUITY"] / df["AMT_INCOME_TOTAL"].replace(0, np.nan)

    if {"AMT_ANNUITY", "AMT_CREDIT"}.issubset(df.columns):
        df["CREDIT_TERM"] = df["AMT_ANNUITY"] / df["AMT_CREDIT"].replace(0, np.nan)

    if {"DAYS_EMPLOYED", "DAYS_BIRTH"}.issubset(df.columns):
        df["DAYS_EMPLOYED_PERCENT"] = df["DAYS_EMPLOYED"] / df["DAYS_BIRTH"].replace(0, np.nan)

    # Sécurisation finale
    num_cols = df.select_dtypes(include=[np.number]).columns
    df[num_cols] = df[num_cols].replace([np.inf, -np.inf], np.nan)
    df[num_cols] = df[num_cols].fillna(df[num_cols].median())

    return df

def make_privacy_binning_reference(df):
    # Bandes d'âge fixes
    age_bins = np.array([18, 25, 30, 35, 40, 45, 50, 55, 60, 70, 120], dtype=float)

    # Bandes de revenu basées sur le réel
    income = pd.to_numeric(df["AMT_INCOME_TOTAL"], errors="coerce")
    income = income.fillna(income.median())
    income_bins = np.unique(np.quantile(income, q=np.linspace(0, 1, 11)))

    if len(income_bins) < 2:
        income_bins = np.array([income.min() - 1, income.max() + 1], dtype=float)

    income_bins[0] -= 1e-9
    income_bins[-1] += 1e-9

    return age_bins, income_bins

def build_privacy_view(df, age_bins, income_bins):
    out = df.copy()

    age_years = pd.to_numeric(out["DAYS_BIRTH"], errors="coerce") / 365.0
    income = pd.to_numeric(out["AMT_INCOME_TOTAL"], errors="coerce")
    income = income.fillna(np.nanmedian(income))

    out["AGE_BAND"] = pd.cut(
        age_years,
        bins=age_bins,
        include_lowest=True,
        right=False
    ).astype(str)

    out["INCOME_BAND"] = pd.cut(
        income,
        bins=income_bins,
        include_lowest=True,
        right=False,
        duplicates="drop"
    ).astype(str)

    for c in ["CODE_GENDER", "NAME_FAMILY_STATUS", "NAME_EDUCATION_TYPE", "FLAG_OWN_REALTY"]:
        out[c] = out[c].astype(str).fillna("MISSING")

    out[TARGET_COL] = pd.to_numeric(out[TARGET_COL], errors="coerce").fillna(0).round().astype(int)

    return out[QI_COLS + [TARGET_COL]]

def privacy_report(df_priv, quasi_cols, sensitive_col=TARGET_COL, k=10, l=2):
    grp = (
        df_priv
        .groupby(quasi_cols, dropna=False)[sensitive_col]
        .agg(group_size="size", l_diversity="nunique")
        .reset_index()
    )

    total_rows = int(grp["group_size"].sum())

    report = {
        "n_groups": int(len(grp)),
        "min_group_size": int(grp["group_size"].min()),
        "median_group_size": float(grp["group_size"].median()),
        "pct_groups_below_k": float(100 * (grp["group_size"] < k).mean()),
        "pct_rows_below_k": float(100 * grp.loc[grp["group_size"] < k, "group_size"].sum() / total_rows),
        "min_l_diversity": int(grp["l_diversity"].min()),
        "pct_groups_below_l": float(100 * (grp["l_diversity"] < l).mean()),
        "pct_rows_below_l": float(100 * grp.loc[grp["l_diversity"] < l, "group_size"].sum() / total_rows),
    }

    return report, grp

def filter_k_l_safe_rows(df_raw, age_bins, income_bins, quasi_cols, sensitive_col=TARGET_COL, k=10, l=2):
    priv = build_privacy_view(df_raw, age_bins, income_bins)

    grp = (
        priv
        .groupby(quasi_cols, dropna=False)[sensitive_col]
        .agg(group_size="size", l_diversity="nunique")
        .reset_index()
    )

    merged = priv.merge(grp, on=quasi_cols, how="left")
    mask = (merged["group_size"] >= k) & (merged["l_diversity"] >= l)

    return df_raw.loc[mask.values].reset_index(drop=True)

# 7. Differential Privacy simple avant diffusion
#    Important : c'est une couche pratique, pas un entraînement DP end-to-end
# ============================================================
try:
    from diffprivlib.mechanisms import Laplace
    HAS_DIFFPRIVLIB = True
except Exception:
    HAS_DIFFPRIVLIB = False

def laplace_dp_release(df, cols, epsilon=1.0, clip_quantiles=(0.01, 0.99), seed=42):
    """
    Mécanisme laplacien simple colonne par colonne.
    Pragmatique pour une première diffusion interne.
    """
    out = df.copy()
    rng = np.random.default_rng(seed)

    if len(cols) == 0:
        return out

    eps_per_col = epsilon / len(cols)

    for i, c in enumerate(cols):
        if c not in out.columns:
            continue

        x = pd.to_numeric(out[c], errors="coerce").astype(float)

        # bornage pour réduire la sensibilité
        lower = float(np.nanquantile(x, clip_quantiles[0]))
        upper = float(np.nanquantile(x, clip_quantiles[1]))

        x = np.clip(x, lower, upper)
        sensitivity = max(upper - lower, 1e-6)

        if HAS_DIFFPRIVLIB:
            mech = Laplace(
                epsilon=eps_per_col,
                sensitivity=sensitivity,
                random_state=seed + i
            )
            noisy = np.array([
                mech.randomise(v) if pd.notna(v) else np.nan
                for v in x
            ])
        else:
            scale = sensitivity / max(eps_per_col, 1e-12)
            noisy = x + rng.laplace(loc=0.0, scale=scale, size=len(x))

        noisy = np.clip(noisy, lower, upper)
        out[c] = noisy

    return out

# Version diffusée = d'abord filtre k/l puis bruit DP sur quelques numériques

## Version B — machine verrouillée, sans packages spécialisés de synthèse

Là, je préfère être nette : **un vrai CTGAN / TVAE sans SDV ou sans code PyTorch long et fragile, ce n’est pas raisonnable**. En revanche, tu peux faire une **version de secours sérieuse** avec :

* le **même nettoyage**,
* la **même partie confidentialité**,
* et un **benchmark bayésien simple de type Naive Bayes génératif**, qui tourne avec `numpy/pandas/sklearn` seulement.

Tu reprends **exactement** les fonctions de préparation et de confidentialité de la version A, puis tu remplaces seulement la partie générateurs par ce bloc :

# VERSION DE SECOURS : générateur bayésien simple sans packages spécialisés

import numpy as np
import pandas as pd

def fit_naive_bayes_synthesizer(df, target_col="TARGET", categorical_cols=None):
    df = df.copy()

    if categorical_cols is None:
        categorical_cols = [
            c for c in df.columns
            if c == target_col
            or pd.api.types.is_object_dtype(df[c])
            or pd.api.types.is_string_dtype(df[c])
        ]

    categorical_cols = set(categorical_cols)

    model = {
        "target_col": target_col,
        "target_probs": df[target_col].astype(str).value_counts(normalize=True).sort_index().to_dict(),
        "features": {}
    }

    for col in df.columns:
        if col == target_col:
            continue

        if col in categorical_cols:
            domain = sorted(df[col].astype(str).fillna("MISSING").unique().tolist())
            cond_probs = {}

            for y, g in df.groupby(target_col):
                vc = g[col].astype(str).fillna("MISSING").value_counts(normalize=True)
                probs = {v: float(vc.get(v, 1e-9)) for v in domain}
                s = sum(probs.values())
                probs = {k: v / s for k, v in probs.items()}
                cond_probs[str(y)] = probs

            model["features"][col] = {
                "type": "categorical",
                "domain": domain,
                "conditional_probs": cond_probs,
            }

        else:
            stats = {}
            for y, g in df.groupby(target_col):
                s = pd.to_numeric(g[col], errors="coerce")
                s = s.fillna(s.median())
                stats[str(y)] = {
                    "mean": float(s.mean()),
                    "std": float(max(s.std(ddof=0), 1e-6)),
                }

            model["features"][col] = {
                "type": "gaussian",
                "conditional_stats": stats,
            }

    return model

def sample_naive_bayes_synthesizer(model, n_rows, seed=42):
    rng = np.random.default_rng(seed)

    target_values = list(model["target_probs"].keys())
    target_probs = np.array([model["target_probs"][v] for v in target_values], dtype=float)
    target_probs = target_probs / target_probs.sum()

    targets = rng.choice(target_values, size=n_rows, p=target_probs)

    out = {model["target_col"]: targets}

    for col, spec in model["features"].items():
        values = []

        for y in targets:
            if spec["type"] == "categorical":
                domain = spec["domain"]
                probs = np.array([spec["conditional_probs"][str(y)][v] for v in domain], dtype=float)
                probs = probs / probs.sum()
                values.append(rng.choice(domain, p=probs))
            else:
                mu = spec["conditional_stats"][str(y)]["mean"]
                sigma = spec["conditional_stats"][str(y)]["std"]
                values.append(rng.normal(mu, sigma))

        out[col] = values

    out_df = pd.DataFrame(out)
    out_df["TARGET"] = pd.to_numeric(out_df["TARGET"], errors="coerce").fillna(0).round().astype(int)

    return out_df

# Utilisation

## Temps de run à prévoir

Ordre de grandeur réaliste pour **50 000 lignes / 21 colonnes** :

* préparation + nettoyage : **1 à 3 min**
* **CTGAN, 30 epochs** : **20 à 60 min CPU** ; souvent **5 à 15 min GPU**
* **TVAE, 30 epochs** : **10 à 30 min CPU** ; souvent **3 à 10 min GPU**
* **modèle bayésien** : **3 à 15 min**
* confidentialité + exports : **moins de 5 min**

À **100 epochs**, tu peux grossièrement multiplier les temps CTGAN/TVAE par 3. Les docs SDV indiquent d’ailleurs que `epochs` augmente le temps d’entraînement et qu’il n’existe pas de réglage universel ; la doc mentionne aussi 300 epochs par défaut pour CTGAN et TVAE. ([docs.sdv.dev][2])
